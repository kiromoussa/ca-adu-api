# ADU Atlas - GIS / ArcGIS ingestion worker dependencies.
# Covers ingestion/arcgis (ArcGIS REST client) and ingestion/gis (source
# ingesters + run entrypoint). Pinned to versions current as of 2026-07.
#
# Install:  pip install -r ingestion/gis/requirements.txt

# HTTP client for the ArcGIS REST calls (metadata, query, pagination).
httpx>=0.27,<0.29

# Retry / backoff on transient ArcGIS failures (429 / 5xx / transport errors).
tenacity==9.1.2

# Direct PostGIS access. Geometry is built with ST_GeomFromGeoJSON /
# ST_SetSRID / ST_Multi / ST_Centroid and areas with ST_Area(ST_Transform(...)),
# which PostgREST cannot express - so we use psycopg (v3) against SUPABASE_DB_URL.
psycopg[binary]>=3.1,<4

# Geometry helpers for optional client-side normalization / validation.
shapely>=2.0,<3
# Projection support (equal-area area computation is done in PostGIS via SRID
# 3310; pyproj is available for any client-side coordinate work / validation).
pyproj>=3.6,<4

# supabase-py is available for optional Storage uploads of raw snapshot bytes.
supabase==2.15.3

# Local .env convenience for development (no-op in production).
python-dotenv==1.1.1
